    # Header
    import clr
    clr.AddReference("Interop.SolidEdge")
    clr.AddReference("SolidEdge.Community")
    clr.AddReference("System.Runtime.InteropServices")
     
    # Import Solid Edge Classes
    import SolidEdgeFramework as seFramework
    import SolidEdgePart as sePart
    import SolidEdgeCommunity as seCommunity
    import System.Runtime.InteropServices as SRI
    from System import Console, Array
     
    # Solid Edge Objects
    seApplication = None
    seDocuments = None
    sePartDocument = None
     
    # Start tasks here
    try:
        seCommunity.OleMessageFilter.Register()
        seApplication = seCommunity.SolidEdgeUtils.Connect(True)
        seDocuments = seApplication.Documents
        sePartDocument = seDocuments.Add("SolidEdge.PartDocument")
        
    except :
        print("Exception")
    finally:
        seCommunity.OleMessageFilter.Ungister()
